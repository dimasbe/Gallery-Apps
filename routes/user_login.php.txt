<?php

use App\Http\Controllers\AplikasiController;
use Illuminate\Support\Facades\Route;

route::middleware('auth')->prefix('user-login')->group(function (){
    route::get('aplikasi', [AplikasiController::class, 'index'])->name('user_login.aplikasi.index');
    route::get('aplikasi/tambah', [AplikasiController::class, 'create'])->name('user_login.aplikasi.create');
    route::post('aplikasi', [AplikasiController::class, 'store'])->name('user_login.aplikasi.store');
    Route::get('aplikasi/{aplikasi}', [AplikasiController::class, 'show'])->name('user_login.aplikasi.show');
    Route::get('aplikasi/{aplikasi}/edit', [AplikasiController::class, 'edit'])->name('user_login.aplikasi.edit');
    Route::put('aplikasi/{aplikasi}', [AplikasiController::class, 'update'])->name('user_login.aplikasi.update');
    Route::delete('aplikasi/{aplikasi}', [AplikasiController::class, 'destroy'])->name('user_login.aplikasi.destroy');
});
